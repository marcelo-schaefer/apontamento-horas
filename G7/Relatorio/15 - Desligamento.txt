SELECT 
IFNULL(ds.numeroSolicitacao, '') as "Numero solicitação",
IFNULL(ds.nomeProcesso, '') as "Nome do processo",
IFNULL(ds.dataSolicitacao, '') as "Data solicitação",
IFNULL(ds.matriculaSolicitante, '') as "Matrícula Solicitante",
IFNULL(ds.nomeSolicitante, '') as "Nome Solicitante",
IFNULL(ds.empresaSolicitante, '') as "Empresa Solicitante",
IFNULL(ds.filialSolicitante, '') as "Filail Solicitante",    
IFNULL(ds.postoSolicitante, '') as "Posto de Trabalho Solicitante",
IFNULL(ds.matriculaColaborador, '') as "Matrícula Colaborador Desligado",
IFNULL(ds.nomeColaborador, '') as "Nome Colaborador Desligado",  
IFNULL(ds.postoColaborador, '') as "Posto de Trabalho Colaborador Desligado",
IFNULL(ds.centroCustoColaborador, '') as "Centro de Custo Colaborador Desligado",
IFNULL(ds.dataAdmissaoColaborador, '') as "Data de Admissão Colaborador Desligado",
IFNULL(ds.tipoDemissao, '') as "Tipo de demissão",  
IFNULL(ds.dDataDemissao, '') as "Data de demissão",
IFNULL(ds.motivoDesligamento, '') as "Motivo do desligamento",
IFNULL(ds.causaDemissao, '') as "Causa da demissão",
IFNULL(ds.statusSolicitacao, '') as "Status"
FROM "platform-workflow"."bpm_desligamento" ds 
WHERE ds.start_date >= ${"platform-workflow"."bpm_desligamento"."start_date":  Data_inicial}
AND ds.start_date <= ${"platform-workflow"."bpm_desligamento"."start_date":  Data_final}
